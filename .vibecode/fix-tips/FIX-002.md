### Fix C2 — Thêm hàm `import_phbc_excel` vào `etl/importer.py`

Thêm vào cuối file:

```python
def import_phbc_excel(db_path, excel_path, import_batch=None):
    """Import dữ liệu Phát hành báo chí từ Excel vào bảng transactions_phbc."""
    import pandas as pd
    import sqlite3
    from datetime import datetime
    
    batch = import_batch or datetime.now().strftime("%Y%m%d_%H%M%S")
    
    try:
        # Đọc file (hỗ trợ cả .xls và .xlsx)
        ext = str(excel_path).lower()
        df = pd.read_excel(excel_path, engine='xlrd' if ext.endswith('.xls') else 'openpyxl')
    except Exception as e:
        return {"success": False, "rows": 0, "message": f"Lỗi đọc file: {e}"}
    
    # Chuẩn hóa tên cột (lowercase, strip)
    df.columns = [str(c).strip().lower().replace(" ", "_") for c in df.columns]
    
    # Map cột linh hoạt
    col_map = {}
    for c in df.columns:
        if 'buu_cuc' in c or 'ma_bc' in c or c == 'ma': col_map['ma_buu_cuc'] = c
        elif 'doanh_thu' in c or c in ('dt', 'revenue'): col_map['doanh_thu'] = c
        elif 'thang' in c: col_map['thang_du_lieu'] = c
        elif 'nam' in c or 'year' in c: col_map['nam_du_lieu'] = c
    
    required = ['ma_buu_cuc', 'doanh_thu', 'thang_du_lieu', 'nam_du_lieu']
    missing = [r for r in required if r not in col_map]
    if missing:
        return {"success": False, "rows": 0, "message": f"Thiếu cột: {', '.join(missing)}"}
    
    df = df.rename(columns={v: k for k, v in col_map.items()})
    df = df[required].dropna(subset=['ma_buu_cuc'])
    
    # Chuẩn hóa thang_du_lieu thành T01, T02...
    def fmt_thang(v):
        try: return f"T{int(str(v).replace('T','').strip()):02d}"
        except: return str(v)
    df['thang_du_lieu'] = df['thang_du_lieu'].apply(fmt_thang)
    df['nam_du_lieu'] = df['nam_du_lieu'].astype(int)
    df['doanh_thu'] = pd.to_numeric(df['doanh_thu'], errors='coerce').fillna(0)
    df['ma_buu_cuc'] = df['ma_buu_cuc'].astype(str).str.strip().str.upper()
    df['import_batch'] = batch
    
    conn = sqlite3.connect(str(db_path))
    try:
        df.to_sql('transactions_phbc', conn, if_exists='append', index=False,
                  method='multi', chunksize=500)
        conn.commit()
        # Ghi log
        conn.execute("""INSERT OR IGNORE INTO import_log 
            (filename, service_type, rows_imported, import_batch, imported_at)
            VALUES (?, ?, ?, ?, datetime('now','localtime'))""",
            (str(excel_path).split('/')[-1].split('\\')[-1], 'PHBC', len(df), batch))
        conn.commit()
        return {"success": True, "rows": len(df), "message": f"✅ Đã nạp {len(df)} dòng PHBC"}
    except Exception as e:
        conn.rollback()
        return {"success": False, "rows": 0, "message": f"Lỗi ghi DB: {e}"}
    finally:
        conn.close()
```
